  int age;
  public Person(int age) {
    this.age = age;
  }
  public String toString() {

  }
}

class PersonComparator implements Comparator<Person>{
  public int compare(Person m1, Person m2) {
    return m1.age > m2.age ? -1 : m1.age < m2.age ? 1 : 0;
  }
}
