import java.util.Scanner;

class Node{
    public Node(Node pre,String name,Node next){
        this.name = name;
        this.pre = pre;
        this.next = next;
    }
    Node pre = null;
    String name;
    Node next = null;
}

class ChainList{
    static int nodeNum = 0;
    static Node root = new Node(null,"root",null);
    public static Node add(Node pre,String name){
        pre.next = new Node(pre,name,root.next);
        nodeNum++;
        return pre.next;
    }

    public static void remove(Node toRemove){
        toRemove.pre.next = toRemove.next;
        toRemove.next.pre = toRemove.pre;
        nodeNum--;
    }
    public Node getNextNode(Node root){
        return root.next;
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int n = scan.nextInt();
        scan.nextLine();
        String[] s = scan.nextLine().split(",");
        ChainList nameList= new ChainList();
        Node pre = ChainList.root;
        for(String name : s){
            pre = ChainList.add(pre,name);
        }
        pre.next.pre = pre;

        Node toRemove = ChainList.root;
        StringBuffer output = new StringBuffer();
        while (ChainList.nodeNum!=0){
            for(int i=0;i<n;i++){
                toRemove = nameList.getNextNode(toRemove);
            }
            output.append(toRemove.name);
            if(ChainList.nodeNum != 1){
                output.append(",");
            }
            ChainList.remove(toRemove);
        }
        System.out.println(output);
    }
}
